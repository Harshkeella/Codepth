        function autoResize(el) {
            el.style.height = 'auto';
            el.style.height = el.scrollHeight + 'px';
        }

        function handleKeydown(e, cellId) {
            if (e.key === 'Tab') {
                e.preventDefault();
                const ta = e.target;
                const s = ta.selectionStart, end = ta.selectionEnd;
                ta.value = ta.value.substring(0, s) + '    ' + ta.value.substring(end);
                ta.selectionStart = ta.selectionEnd = s + 4;
            }
            if ((e.ctrlKey || e.metaKey) && e.key === 'Enter') {
                e.preventDefault();
                if (cellId === 'cell1') runCell1();
                else runCell2();
            }
        }

        async function callAPI(code, btnId, counterId, outputBoxId, labelId, textId, countVar) {
            const btn         = document.getElementById(btnId);
            const cellCounter = document.getElementById(counterId);
            const outputBox   = document.getElementById(outputBoxId);
            const outputLabel = document.getElementById(labelId);
            const outputText  = document.getElementById(textId);

            if (!code || btn.disabled) return;

            outputBox.className     = 'output-container-cell visible';
            outputLabel.textContent = 'Running...';
            outputText.textContent  = '';
            btn.disabled = true;
            btn.classList.add('loading');
            cellCounter.textContent = 'In [*]:';

            try {
                // ✅ Correct Groq endpoint (OpenAI-compatible)
                const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${API_KEY}`
                    },
                    body: JSON.stringify({
                        model: 'llama3-70b-8192',
                        max_tokens: 1000,
                        messages: [
                            {
                                role: 'system',
                                content: `You are a Python interpreter simulator.
When given Python code, return ONLY the exact output that would be printed to stdout.
- No explanations, no markdown, no backticks
- If code has print(), show exactly what it prints
- If code raises an error, show the exact Python traceback
- If there is no output, return the text: (no output)
Nothing else. Just the raw output.`
                            },
                            {
                                role: 'user',
                                content: code
                            }
                        ]
                    })
                });

                if (!response.ok) {
                    const err = await response.json().catch(() => ({}));
                    throw new Error(err?.error?.message || 'HTTP ' + response.status);
                }

                const data = await response.json();
                countVar.n++;

                // ✅ Groq returns OpenAI format: choices[0].message.content
                const result  = (data.choices?.[0]?.message?.content || '(no output)').trim();
                const isError = result.includes('Traceback') || result.toLowerCase().includes('error:');

                cellCounter.textContent = `In [${countVar.n}]:`;
                outputLabel.textContent = `Out [${countVar.n}]:`;
                outputText.textContent  = result;
                outputBox.className     = 'output-container-cell visible' + (isError ? ' error-output' : '');

            } catch (err) {
                countVar.n++;
                cellCounter.textContent = `In [${countVar.n}]:`;
                outputLabel.textContent = 'Error:';
                outputText.textContent  = '❌ ' + err.message;
                outputBox.className     = 'output-container-cell visible error-output';
            }

            btn.disabled = false;
            btn.classList.remove('loading');
        }

        const c1 = { n: 0 };
        function runCell1() {
            const code = document.getElementById('pythonCode1').value.trim();
            callAPI(code, 'runBtn1', 'cellCounter1', 'outputContainer1', 'outputLabel1', 'outputText1', c1);
        }

        const c2 = { n: 0 };
        function runCell2() {
            const code = document.getElementById('pythonCode2').value.trim();
